// --- MOB RANDOM WALKER --- // Poruszanie obrazka "mob-sectrumgreen.png" po dole ekranu // Stała prędkość + losowe zmiany kierunku document.addEventListener("DOMContentLoaded", () => { const mob = document.getElementById("ludzik"); const speed = 2; // px per frame let direction = Math.random() < 0.5 ? -1 : 1; function moveMob() { const screenWidth = window.innerWidth; const mobWidth = mob.offsetWidth; let x = mob.offsetLeft; // Losowa zmiana kierunku (1% szansy na klatkę) if (Math.random() < 0.01) { direction = Math.random() < 0.5 ? -1 : 1; } // Ruch x += direction * speed; // Odbijanie od krawędzi if (x <= 0) { x = 0; direction = 1; } if (x >= screenWidth - mobWidth) { x = screenWidth - mobWidth; direction = -1; } mob.style.left = x + "px"; requestAnimationFrame(moveMob); } moveMob(); });
